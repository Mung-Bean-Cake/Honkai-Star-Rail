IPC_Idol = {
	category = specialist
	condition_string = SPECIALIST_JOB_TRIGGER
	building_icon = building_holo_theatres
	clothes_texture_index = 3

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible = {
		entertainer_job_check_trigger = yes
	}

	resources = {
		category = planet_entertainers
		category = planet_Idols
		produces = {
			unity = 2
			energy = 4
			society_research = 3
		}
		produces = {
			trigger = {
				has_civic = civic_LightIdol_Institution
			}
			unity = 3
		}
		produces = {
			trigger = {
				has_civic = civic_DarkIdol_Institution
			}
			energy = 4
			society_research = 3
		}
		upkeep = {
			consumer_goods = 1
		}
		upkeep = {
			trigger = {
				OR = {
					has_civic = civic_LightIdol_Institution
					has_civic = civic_DarkIdol_Institution
				}
			}
			consumer_goods = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_add = 12
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_civic = civic_LightIdol_Institution
					has_civic = civic_DarkIdol_Institution
				}
			}
		}
		planet_amenities_add = 6
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}
	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		trade_value_add = 4
		trade_value_mult = 0.05
		mult = value:scripted_modifier_councilor_level_mult|MODIFIER|pop_job_trade_mult|TYPE|councilor_corporate_hedonism|
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			has_civic = civic_LightIdol_Institution
			has_civic = civic_DarkIdol_Institution
		}
		trade_value_add = 2
		trade_value_mult = 0.05
		mult = value:scripted_modifier_councilor_level_mult|MODIFIER|pop_job_trade_mult|TYPE|councilor_corporate_hedonism|
	}

	triggered_planet_modifier = {
		potential = {
			has_councilor = { COUNCILOR = "councilor_corporate_hedonism" }
		}

		trade_value_add = @councilor_corporate_hedonism_entertainer_trade_value
		mult = value:scripted_modifier_councilor_level_mult|MODIFIER|pop_job_trade_mult|TYPE|councilor_corporate_hedonism|
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_civic = civic_DarkIdol_Institution }
		}
		modifier = {
			pop_growth_speed = 0.02
			planet_livestock_produces_mult = 0.02
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_civic = civic_LightIdol_Institution }
		}
		modifier = {
			planet_jobs_worker_produces_mult = 0.02
			planet_stability_add = 2
		}
	}

	weight = {
		weight = @high_prio_specialist_job_weight
		mult = value:job_weights_modifier|JOB|entertainer|RESOURCE|unity|FACTOR|0.25|
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 3
			has_trait = trait_robot_domestic_protocols
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
		}
		modifier = {
			factor = 1.5
			has_job = Idol
		}
		modifier = {
			factor = 0.1
			is_domestic_servant = yes
		}
		modifier = {
			factor = 2
			NOT = { has_job = Idol }
			exists = planet
			planet = {
				free_amenities < 10
				has_available_jobs = Idol
			}
		}
		modifier = {
			add = 500
			exists = planet
			planet = {
				has_modifier = artistic_boom
			}
		}
	}
}
