change_planet_to_pc_shield_special = {
    random_system_planet = {
       limit = {
            NOR = {
                is_colonizable = yes
                is_colony = yes
                is_star = yes
                is_planet_class = pc_shielded_white
                is_planet_class = pc_shielded_red
                is_planet_class = pc_shielded_yellow
                is_planet_class = pc_astral_scar
            }
        }
        change_pc = $planet_class$
        set_planet_flag = $planet_flag$
    }
}
